     172   static List<Festival> sortByDate(List<Festival> festivals, [DateTime? now]) {
     173     final currentDate = now ?? DateTime.now();
     174     final sorted = List<Festival>.from(festivals);
     175     
     176     sorted.sort((a, b) {
     177       final statusA = a.getBasicStatus(currentDate);
     178       final statusB = b.getBasicStatus(currentDate);
     179       
     180       // Priority: live > upcoming > past
     181       if (statusA == FestivalStatus.live && statusB != FestivalStatus.live) {
     182         return -1;
     183       }
     184       if (statusB == FestivalStatus.live && statusA != FestivalStatus.live) {
     185         return 1;
     186       }
     187       
     188       if (statusA == FestivalStatus.upcoming && statusB == FestivalStatus.past) {
     189         return -1;
     190       }
     191       if (statusB == FestivalStatus.upcoming && statusA == FestivalStatus.past) {
     192         return 1;
     193       }
     194       
     195       // Within same status, sort by date
     196       if (statusA == FestivalStatus.upcoming && statusB == FestivalStatus.upcoming) {
     197         // Upcoming: soonest first. Festivals with null startDate are sorted last.
     198         if (a.startDate == null && b.startDate == null) return 0;
